#include<stdio.h>
#include<string.h>
#include<ctype.h>
#define MAX 256
void check_phone_str(char* phone) {
	int T = 0;
	if (phone[0] == '0' && phone[1] == '1' && phone[2] == '0') {
		for (int i = 3; i <= 10; i++) {
			if (phone[i] >= '0' && phone[i] <= '9')
				T++;
		}
	}
	if (T == 8)
		printf("%s는 유효한 전화번호입니다.\n", phone);
	else
		printf("잘못 입력했습니다. 01012345678 형식으로 입력하세요.\n");

}
int main(void) {
	char phone[MAX];

	for (;;) {
		printf("전화번호(. 입력 시 종료)? ");
		scanf_s("%s", &phone, MAX);
		check_phone_str(phone);
		if (*phone == '.')
			break;
	}
}
