#include<stdio.h>
void find_array(int *arr, int a) {
	for (int i = 0; i < 10; i++) {
		if (arr[i] == a) {
			printf("%d는 %d번째 원소입니다.\n", a, i+1);
		}
		else
			printf("올바른 숫자를 입력하세요.\n");
		break;
	}
}
int main(void) {
	int arr[10] = { 23,45,62,12,99,83,23,50,72,37 };
	int a;

	for (int i = 0; i < 10; i++) {
		printf("%4d", arr[i]);
	}
	printf("\n찾을 값? ");
	scanf_s("%d", &a);
	find_array(arr, a);
}
