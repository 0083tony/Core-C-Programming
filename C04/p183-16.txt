#include<stdio.h>
int main(void) {
	int r, g, b;
	int color = 0;

	printf("red? ");
	scanf_s("%d", &r);
	printf("green? ");
	scanf_s("%d", &g);
	printf("blue? ");
	scanf_s("%d", &b);

	if (r > 255) {
		r = 0;
	}
	if (g > 255) {
		g = 0;
	}
	if (b > 255) {
		b = 0;
	}
	color = color | (b << 16);
	color = color | (g << 8);
	color = color | r;

	printf("RGB 트루컬러: %06x", color);
}
