#include<stdio.h>
#include<stdlib.h>
int divisors(int a) {
	int i, count = 0;
	
	printf("%d의 약수: ",a);

	for (i = 1; i <= a; i++) {
		if (a % i == 0) {
			count++;
			printf(" %d ", i);
		}
		
	}
	printf("==> 총 %d개\n", count);
	return 0;
}
int main(void) {
	int i;
	
	for (i = 0; i < 3; i++) {
		int a = rand() % 1000;
		divisors(a);
	}
}
