#include<stdio.h>
int is_even(int a) {
	int even = 0;

	if (a % 2 == 0) {
		even++;
	}
	return even;
}
int is_odd(int a) {
	int odd = 0;

	if (a % 2 == 1) {
		odd++;
	}
	return odd;
}
int main(void) {
	int a;
	int odd = 0;
	int even = 0;

	printf("정수를 빈칸으로 구분해서 입력하세요.(마지막에 0 입력)\n");

	for (;;) {
		scanf_s("%d", &a);

		if (is_odd(a) == 1) {
			odd++;
		}
		else {
			even++;
		}
		if (a == 0) {
			printf("입력받은 정수 중 짝수는 %d개, 홀수는 %d개입니다.", even - 1, odd);
		}
	}
}
