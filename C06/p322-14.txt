#include<stdio.h>
int check_date (int year, int month, int date) {
	int err = 0;
	
	if ((year % 4 == 0) && (year % 100 != 0) || (year % 400 == 0)) {
		if (month < 0 || month > 12){
			printf("잘못 입력하셨습니다. 유효한 날짜를 입력하세요.\n");
			err = err + 1;
		}
		if ((month % 2 == 0 && date > 30) || (month == 2 && date > 29) || (month % 1 && date > 31)){
			printf("잘못 입력하셨습니다. 유효한 날짜를 입력하세요.\n");
			err = err + 1;
		}			
	}	else if (month < 0 || month > 12) {
			printf("잘못 입력하셨습니다. 유효한 날짜를 입력하세요.\n");
			err = err + 1;
			
			if ((month % 2 == 0 && date > 30) || (month == 2 && date > 28) || (month % 1 && date > 31)){
			printf("잘못 입력하셨습니다. 유효한 날짜를 입력하세요.\n");
			err = err + 1;
			}
		}
	return err;
}
int main(void) {
	int year, month, date;
	int err = 0;
	
	for (;;) {
		printf("날짜 (연 월 일)? ");
		scanf_s("%d %d %d", &year, &month, &date);
				
		if (check_date(year, month, date) == 0) {
			printf("입력한 날짜는 %d년 %d월 %d일 입니다.\n", year, month, date);
			break;
		}
	}
}
