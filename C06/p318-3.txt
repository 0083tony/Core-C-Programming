#include<stdio.h>
#include<math.h>
double distance(int a, int b, int c, int d){
	float dis = 0;
	
	dis = sqrt(pow(c - a, 2) + pow(d - b, 2));
	return dis;
}
int main(void) {
	int a, b, c, d;
	
	printf("직선의 시작점 좌표? ");
	scanf_s("%d %d", &a, &b);
	printf("직선의 끝점 좌표? ");
	scanf_s("%d %d", &c, &d);

	printf("(%d,%d)~(%d,%d) 직선의 길이: %f", a, b, c, d, distance(a, b, c, d));
}
