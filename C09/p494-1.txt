#include<stdio.h>
#include<string.h>
#include<ctype.h>
int main(void) {
	char str[256];
	int a = 0;
	
	printf("문자열? ");
	scanf_s("%[^\n]", str, 256);

	int n = strlen(str);

	for (int i = 0; i < n; i++) {
		if (isspace(str[i]))
			a++;
	}

	printf("공백 문자의 개수: %d", a);
}
