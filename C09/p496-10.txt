#define _CRT_SECURE_NO_WARNINGS 
#include<stdio.h>
#include<string.h>
#include<ctype.h>
#define MAX 256
int main(void) {
	char name[MAX] = "";
	
	
	for (;;) {
		printf("영문 이름? ");
		scanf("%[^\n]s", name);
		getchar();
		if (*name == '.') {
			break;
		}
		printf("이니셜: ");
		_strupr(name);

		for (int i = 0; i < strlen(name); i++) {
			if (i == 0) {
				printf("%c", name[0]);
			}
			if (isblank(name[i])) {
				printf("%c", name[i + 1]);
			}
		}
		printf("\n");
	}
	
}
