#include<stdio.h>
float calculator(float a, float b, char c) {
	float n = 0;

	if (c == '+') {
		n = a + b;
	}
	if (c == '-') {
		n = a - b;
	}
	if (c == '*') {
		n = a * b;
	}
	if (c == '/') {
		n = a / b;
	}
	return n;
}
int main(void) {
	float a = 0, b = 0;
	char c;

	for (;;) {
		printf("수식 (0 0 0 입력 시 종료)? ");
		scanf_s("%f %c %f", &a, &c, 30,&b);

		if (a == 0 && b == 0 && c == '0') {
			break;
		}
		printf("%f\n", calculator(a, b, c));
	}
}
