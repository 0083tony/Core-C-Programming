#include<stdio.h>
void add_to_multiset(int* arr, int i, int a) {
	for (int j = 0; j <= i; j++) {
		printf("%d ", arr[j]);
	}
}
int main(void) {
	int arr[10];
	int a;

	for (int i = 0; i < 10; i++) {
		printf("배열에 추가할 원소? ");
		scanf_s("%d", &a);
		arr[i] = a;
		add_to_multiset(arr,i, a);
		printf("\n");
	}
}
