#include<stdio.h>
#include<stdlib.h>
int random(int add) {
	int random = 0;

	for (int i = 0; i < 10; i++) {
		random = rand() % 100;
		printf("%5d", random);
		add = random + add;
	}
	return add;
}
int main(void) {
	int add = 0;

	printf("0 ~ 99사이의 임의의 정수를 10개 생성해서 합계를 구합니다.\n");
	printf("\n합계: %d", random(add));
}
