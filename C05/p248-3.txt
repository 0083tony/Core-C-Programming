#include<stdio.h>
#include<math.h>
int main(void) {
	int n;
	int a, b, c, d, e, f;

	printf("거스름돈? ");
	scanf_s("%d", &n);

	n = n - n % 10;

	printf("거스름돈 (10원미만 절사): %d\n", n);

	if (n / 50000 >= 1) {
		a = n / 50000;
		printf("50000원 %d장 \n", a);
		if ((n - 50000*a) / 10000 >= 1) {
			b = (n - 50000*a) / 10000;
			printf("10000원 %d장 \n", b);
			if ((n - 50000 * a - 10000 * b) / 5000 >= 1) {
				c = (n - 50000 * a - 10000 * b) / 5000;
				printf("5000원 %d장 \n", c);
				if ((n - 50000 * a - 10000 * b - 5000 * c) / 1000 >= 1) {
					d = (n - 50000 * a - 10000 * b - 5000 * c) / 1000;
					printf("1000원 %d장 \n", d);
					if ((n - 50000 * a - 10000 * b - 5000 * c - 1000 * d) / 100 >= 1) {
						e = (n - 50000 * a - 10000 * b - 5000 * c - 1000 * d) / 100;
						printf("100원 %d장 \n", e);
						if ((n - 50000 * a - 10000 * b - 5000 * c - 1000 * d - 100 * e) / 10 >= 1) {
							f = (n - 50000 * a - 10000 * b - 5000 * c - 1000 * d - 100 * e) / 10;
							printf("10원 %d장 \n", f);
						}
					}
				}
			}
		}
	}
}
