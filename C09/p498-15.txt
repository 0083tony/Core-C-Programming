int time_to_string(char* time, int a, int h, int m, int s) {

	if (h > 24 || m > 60 || s > 60) {
		return -1;
	}
	sprintf(&time[0], "% d", h);
	sprintf(&time[4], "% d", m);
	sprintf(&time[8], "% d", s);
	time[3] = ':';
	time[7] = ':';

	return a;
}
