#include<stdio.h>
#include<string.h>
#include<ctype.h>
void reverse_string(char* str, int n) {
	char temp;

	for (int i = 0; i < n / 2; i++) {
		temp = str[i];
		str[i] = str[(n - 1) - i];
		str[(n - 1) - i] = temp;
	}
}
int main(void) {
	char str[256];
	int n;

	printf("문자열? ");
	scanf_s("%[^\n]", str, 256);

	n = strlen(str);
	
	reverse_string(str, n);
	printf("변환 후: %s", str);
}
