#include<stdio.h>
int get_red(int a) {
	return a >> 16;
}
int get_green(int a) {
	a = a >> 8;
	a = a & 0xFF;
	return a;
}
int get_blue(int a) {
	a = a & 0xFF;
	return a;
}
int make_rgb(int a) {
	int r, g, b;
	int color = 0;
	
	r = get_red(a);
	g = get_green(a);
	b = get_blue(a);

	color = color | (r << 16);
	color = color | (g << 8);
	color = color | b;

	printf("RGB %X의 보색: %X", a, ~color); // 교제와 동일하게 00FF7F만 출력X FF00FF7F로 출력됨
	
	return 0;
}
int main(void) {
	int a;

	printf("RGB 색상? ");
	scanf_s("%x", &a);

	make_rgb(a);
}
