#include<stdio.h>
int main(void) {
	int arr[5];
	float price, n;

	printf("상품가 5개를 입력하세요:\n");

	for (int i = 0; i < 5; i++) {
		scanf_s("%d", &arr[i]);
	}

	printf("할인율(%%)? ");
	scanf_s("%f", &n);

	for (int i = 0; i < 5; i++) {
		price = arr[i] - (arr[i] * (0.01 * n));
		printf("가격:%8d --> 할인가:%8.0f\n", arr[i], price);
	}
}
