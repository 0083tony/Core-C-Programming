#include<stdio.h>
int main(void) {
	int arr[10];
	int overlap = 0;

	for (int i = 0; i < 10; i++) {
		printf("배열에 추가할 원소? ");
		scanf_s("%d", &arr[i]);
		for (int n = 0; n < i; n++) {
			if (arr[n] == arr[i]) {
				printf("\n해당 원소가 이미 [%d]에 존재합니다.\n", n);
				i--;
			}
		}
		for (int j = 0; j <= i; j++) {
			printf("%4d", arr[j]);
		}
		printf("\n");
	}
}
