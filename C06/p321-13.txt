#include<stdio.h>
int get_days_of_month(int year) {
	if ((year % 4 == 0) && (year % 100 != 0) || (year % 400 == 0)) {
		printf("[ %d년 ]\n", year);
		printf("1월: 31일	2월: 29일	3월: 31일	4월: 30일	5월: 31일	6월: 30일\n");
		printf("7월: 31일	8월: 31일	9월: 30일	10월: 31일	11월: 30일	12월: 31일");
	}
	else {
		printf("[ %d년 ]\n", year);
		printf("1월: 31일	2월: 28일	3월: 31일	4월: 30일	5월: 31일	6월: 30일\n");
		printf("7월: 31일	8월: 31일	9월: 30일	10월: 31일	11월: 30일	12월: 31일");
	}
	return 0;
}
int main(void) {
	int year;

	printf("연도? ");
	scanf_s("%d", &year);

	get_days_of_month(year);
}
