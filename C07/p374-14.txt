#include<stdio.h>
#include<stdlib.h>
#include <time.h> 
void suf(int* arr) {
	srand(time(NULL));
	size_t i; // 오버플로우 오류로 int가 아닌 size_t 사용
	for (i = 0; i < 10; i++) {
		size_t j = i + rand() / (RAND_MAX / (10 - i) + 1);
		int t = arr[j];
		arr[j] = arr[i];
		arr[i] = t;
		printf("%4d", t);
	}
}
int main(void) {
	int arr[10] = { 12,64,80,42,67,9,8,7,56,48 };

	printf("셔플 전: ");
	for (int i = 0; i < 10; i++) {
		printf("%4d", arr[i]);
	}
	printf("\n셔플 후: ");
	suf(arr);
}
