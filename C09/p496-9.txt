#define _CRT_SECURE_NO_WARNINGS // stract 경고로 컴파일에러 방지
#include<stdio.h>
#include<string.h>
#include<ctype.h>
#define MAX 256
int main(void) {
	char name[MAX] = "";
	char extn[MAX] = "";
	char dot[2] = ".";

	printf("파일명? ");
	scanf("%[^\n]s", name);
	getchar();
	printf("확장자? ");
	scanf("%[^\n]s", extn);

	strcat(name, dot);
	strcat(name, extn);

	printf("전체 파일명:  %s", name);
}
