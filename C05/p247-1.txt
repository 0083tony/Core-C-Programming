#include<stdio.h>
int main(void) {
	int l, t, r, b;
	int x, y;
	   	
	printf("선택 영역의 좌상단점 (left, top)? ");
	scanf_s("%d %d", &l, &t);
	printf("선택 영역의 우하단점 (right, botton)? ");
	scanf_s("%d %d", &r, &b);

	printf("점의 좌표 (x, y)");
	scanf_s("%d %d", &x, &y);

	
	if (l <= x && x <= r && t <= y && y <= b) {
		printf("직사각형 모양의 선택 영역 내의 점입니다.");
	}
	else {
		printf("직사각형 모양의 선택 영역 안이 아닙니다.");
		return 0;
	}
	
}
