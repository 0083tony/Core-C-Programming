#include<stdio.h>
int main(void) {
	int x, y;

	printf("점의 좌표 (x, y)? ");
	scanf_s("%d %d", &x, &y);

	if (x > 0 && y > 0) {
		printf("\n1사분면에 있습니다.");
	}
	if (x < 0 && y > 0) {
		printf("\n2사분면에 있습니다.");
	}
	if (x < 0 && y < 0) {
		printf("\n3사분면에 있습니다.");
	}
	if (x > 0 && y < 0) {
		printf("\n4사분면에 있습니다.");
	}
}
