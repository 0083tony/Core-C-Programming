	n = n % 50000;
	b = n / 10000;
	n = n % 10000;
	c = n / 5000;
	n = n % 5000;
	d = n / 1000;
	n = n % 1000;
	e = n / 100;
	n = n % 100;
	f = n / 10;
	n = n % 10;

	printf("50000원 %d장 \n", a);

	printf(" 1000원 %d장 \n", d);
