#include<stdio.h>
#include<string.h>
#include<ctype.h>
#define MAX 256
int main(void) {
	char pass[MAX] = "";
	int a = 0;
	size_t b;

	for (;;) {
		printf("문자열? ");
		scanf_s("%[^\n]s", &pass, MAX);
		
		if (*pass == '.')
			break;
		
		printf("암호 키(정수)? ");
		scanf_s("%d", &a);
		getchar(); // 버퍼에 남아있는 \n을 지우지않으면 scanf %s를 무시하게됨.

		b = strlen(pass);
		for (int i = 0; i < b; i++) {
			pass[i] = pass[i] + a;
		}
		printf("암호화된 문자열: %s\n", pass);
	}
}
