#include<stdio.h>
void fill_array_2D(int (*arr)[5], int *a) {
	for (int i = 0; i < 4; i++) {
		for (int j = 0; j < 5; j++) {
			arr[i][j] = *a;
			printf("%d ", arr[i][j]);
		}
		printf("\n");
	}
}
int main(void) {
	int arr[4][5];
	int a;

	printf("배열의 원소에 저장할 값? ");
	scanf_s("%d", &a);
	
	fill_array_2D(arr, &a);
}
