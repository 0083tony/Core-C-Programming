#include<stdio.h>
int main(void) {
	int arr[11];
	int a;

	printf("1 ~ 12월 까지 핸드폰 요금을 입력하세요.\n");
	
	for (int j = 0; j < 12; j++) {
		scanf_s("%d", &arr[j]);
	}

	for (int i = 0; i < 12; i++) {
		printf("\n%d월 %d: ", i + 1, arr[i]);
		a = arr[i] / 2000;
		
		for (int l = 0; l < a; l++) {
			printf("*");
		}
	}
}
