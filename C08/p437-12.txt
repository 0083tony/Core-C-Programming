#include<stdio.h>
void find_all_in_array(int* arr, int a) {
	int count = 0;
	for (int i = 0; i < 10; i++) {
		if (arr[i] == a) {
			count++;
		}
	}
	printf("찾은 항목은 모두 %d개입니다.\n", count);
	printf("찾은 항목의 인덱스: ");
	for (int i = 0; i < 10; i++) {
		if (arr[i] == a) {
			printf("%3d", i);
		}
	}
}
int main(void) {
	int arr[10] = { 12,45,62,12,99,83,23,12,72,37 };
	int a;

	for (int i = 0; i < 10; i++) {
		printf("%3.d", arr[i]);
	}
	printf("\n찾을 값? ");
	scanf_s("%d", &a);
	find_all_in_array(arr, a);
}
