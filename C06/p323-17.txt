#include<stdio.h>
int get_quadrant(int x,int y) {

	if (x > 0 && y > 0) {
		printf("1사분면에 있습니다.\n");
		return 1;
	}
	if (x < 0 && y > 0) {
		printf("2사분면에 있습니다.\n");
		return 2;
	}
	if (x < 0 && y < 0) {
		printf("3사분면에 있습니다.\n");
		return 3;
	}
	if (x > 0 && y < 0) {
		printf("4사분면에 있습니다.\n");
		return 4;
	}
	return 0;
}
int main(void) {
	int x, y;

	for (;;) {
		printf("점의 좌표 (x, y)? ");
		scanf_s("%d %d", &x, &y);

	if (get_quadrant(x, y) == 0)
		break;
	}

}
